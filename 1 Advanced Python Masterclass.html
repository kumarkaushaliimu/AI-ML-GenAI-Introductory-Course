📋 For Loops: Sequence Iteration

Iterate over lists, tuples, strings, ranges

# Basic for loop fruits = ["apple", "banana", "cherry"] for fruit in fruits: print(f"I like {fruit}") # For loop with range for i in range(5): print(i) # 0, 1, 2, 3, 4 # For loop with enumerate (index + value) for i, fruit in enumerate(fruits): print(f"{i}: {fruit}") # Nested loops (multiplication table) for i in range(1, 4): for j in range(1, 4): print(f"{i}x{j}={i*j}", end=" ") print() # New line

🛑 Loop Control: Break, Continue, Pass

Modify loop behavior

break

Exit loop immediately

for i in range(10): if i == 5: break print(i) # Output: 0,1,2,3,4

continue

Skip to next iteration

for i in range(5): if i == 2: continue print(i) # Output: 0,1,3,4

pass

Placeholder (do nothing)

for i in range(3): if i == 1: pass # TODO: implement else: print(i)

✨ List & Dict Comprehensions

Pythonic way to create collections

List Comprehension

# Traditional squares = [] for x in range(5): squares.append(x**2) # Pythonic squares = [x**2 for x in range(5)] # [0, 1, 4, 9, 16] # With condition evens = [x for x in range(10) if x % 2 == 0] # [0, 2, 4, 6, 8] # Nested matrix = [[i+j for j in range(3)] for i in range(3)]

Dict Comprehension

# Basic squares_dict = {x: x**2 for x in range(5)} # {0:0, 1:1, 2:4, 3:9, 4:16} # With condition scores = {"A":85, "B":90, "C":78} passing = {k:v for k,v in scores.items() if v >= 85} # {'A':85, 'B':90} # From lists keys = ['x', 'y', 'z'] values = [1, 2, 3] d = {k:v for k,v in zip(keys, values)}
✓ Benefit: Faster, more readable, fewer lines of code

⚙️ Functions: Reusable Code Blocks

Define and call functions with parameters

# Function definition def calculate_gpa(scores): """Calculate GPA from scores list.""" total = sum(scores) count = len(scores) return total / count if count > 0 else 0 # Function call student_scores = [85, 90, 78, 92] gpa = calculate_gpa(student_scores) print(f"GPA: {gpa:.2f}") # GPA: 86.25 # Function with multiple parameters def grade_student(name, score): """Return grade for student.""" if score >= 90: grade = 'A' elif score >= 80: grade = 'B' else: grade = 'C' return f"{name}: {grade}" print(grade_student("Alice", 85)) # Alice: B
Components: def keyword, function name, parameters, docstring, body, return statement

📥 Function Arguments: Parameters & Defaults

Default arguments, keyword arguments, *args, **kwargs

# Default arguments def greet(name, greeting="Hello"): return f"{greeting}, {name}!" print(greet("Alice")) # Hello, Alice! print(greet("Bob", "Hi")) # Hi, Bob! # Keyword arguments (any order) def create_profile(name, age, city): return f"{name}, {age}, {city}" print(create_profile(city="NYC", name="Carol", age=28)) # *args - variable positional arguments def sum_all(*numbers): return sum(numbers) print(sum_all(1, 2, 3)) # 6 print(sum_all(1, 2, 3, 4, 5)) # 15 # **kwargs - variable keyword arguments def print_info(**kwargs): for key, value in kwargs.items(): print(f"{key}: {value}") print_info(name="Alice", age=25, city="Delhi")

🔐 Variable Scope: LEGB Rule

Local, Enclosing, Global, Built-in

# Global scope global_var = "Global" def outer(): # Enclosing scope enclosing_var = "Enclosing" def inner(): # Local scope local_var = "Local" print(local_var) # Local print(enclosing_var) # Enclosing print(global_var) # Global print(len("test")) # Built-in inner() outer() # Modifying global counter = 0 def increment(): global counter counter += 1 increment() print(counter) # 1

LEGB Order

  1. Local: Inside function
  2. Enclosing: Outer function
  3. Global: Module level
  4. Built-in: Python built-ins

⚡ Lambda Functions: Anonymous Functions

Single-line functions without def keyword

# Basic lambda add = lambda x, y: x + y print(add(5, 3)) # 8 # Lambda with map numbers = [1, 2, 3, 4, 5] squared = list(map(lambda x: x**2, numbers)) print(squared) # [1, 4, 9, 16, 25] # Lambda with filter numbers = [1, 2, 3, 4, 5, 6] even_nums = list(filter(lambda x: x % 2 == 0, numbers)) print(even_nums) # [2, 4, 6] # Lambda with sorted (key parameter) students = [("Alice", 85), ("Bob", 92), ("Charlie", 78)] sorted_by_score = sorted(students, key=lambda x: x[1]) print(sorted_by_score) # [('Charlie', 78), ('Alice', 85), ('Bob', 92)]
Use cases: Simple operations, sorting keys, map/filter operations, callbacks

🔧 Functional Programming: Map, Filter, Reduce

Process iterables efficiently

map()

Apply function to all items

temps_c = [0, 10, 20, 30] temps_f = list(map( lambda c: c*9/5 + 32, temps_c)) # [32, 50, 68, 86]

filter()

Keep items that pass test

scores = [45, 67, 89, 34, 92] passing = list(filter( lambda x: x >= 60, scores)) # [67, 89, 92]

reduce()

Reduce to single value

from functools import reduce nums = [1, 2, 3, 4, 5] product = reduce( lambda x, y: x*y, nums) # 120

⚠️ Exception Handling: Try-Except-Finally

Gracefully handle errors in code

Try-catch flow diagram
# Basic try-except try: file = open('nonexistent.txt') except FileNotFoundError: print("File not found!") # Multiple exceptions try: num = int("abc") except ValueError: print("Invalid number") except FileNotFoundError: print("File not found") # Try-except-else-finally try: result = 10 / 2 except ZeroDivisionError: print("Cannot divide by zero") else: print(f"Result: {result}") finally: print("Cleanup")

🏗️ Object-Oriented Programming: Classes

Define classes and create objects

# Class definition class Student: """Represents a student.""" def __init__(self, name, student_id, gpa=0.0): """Initialize student.""" self.name = name self.student_id = student_id self.gpa = gpa def introduce(self): """Student introduction.""" return f"I'm {self.name} ({self.student_id}), GPA: {self.gpa}" def update_gpa(self, new_gpa): """Update GPA.""" self.gpa = new_gpa # Create objects (instances) student1 = Student("Alice", "S001", 3.85) student2 = Student("Bob", "S002") print(student1.introduce()) # I'm Alice (S001), GPA: 3.85 student1.update_gpa(3.9) print(student1.gpa) # 3.9
Components: __init__() constructor, self parameter, instance attributes, methods

📊 Class vs Instance Attributes

Shared attributes vs individual attributes

class Dog: species = "Canis familiaris" # Class attribute (shared) def __init__(self, name, age): self.name = name # Instance attribute self.age = age # Instance attribute dog1 = Dog("Buddy", 3) dog2 = Dog("Max", 5) # Instance attributes are different print(dog1.name) # Buddy print(dog2.name) # Max # Class attribute is shared print(Dog.species) # Canis familiaris print(dog1.species) # Canis familiaris (accessed via instance) print(dog2.species) # Canis familiaris # Modifying class attribute Dog.species = "Canis lupus familiaris" print(dog1.species) # Updated for all instances

🌳 Inheritance: Extend Classes

Create subclasses from parent classes

Class inheritance hierarchy
# Parent class class Animal: def __init__(self, name): self.name = name def speak(self): return "Sound" # Child classes class Dog(Animal): def speak(self): return f"{self.name} barks!" class Cat(Animal): def speak(self): return f"{self.name} meows!" # Usage dog = Dog("Buddy") cat = Cat("Whiskers") print(dog.speak()) # Buddy barks! print(cat.speak()) # Whiskers meows!

🔀 Polymorphism: Multiple Forms

Different objects responding to same method call

# Polymorphism example class Shape: def area(self): pass class Circle(Shape): def __init__(self, radius): self.radius = radius def area(self): return 3.14159 * self.radius ** 2 class Rectangle(Shape): def __init__(self, width, height): self.width = width self.height = height def area(self): return self.width * self.height # Polymorphic function def print_areas(shapes): for shape in shapes: print(f"Area: {shape.area():.2f}") shapes = [Circle(5), Rectangle(4, 6)] print_areas(shapes) # Area: 78.54 # Area: 24.00
Benefit: Code reusability, extensibility, clean design
